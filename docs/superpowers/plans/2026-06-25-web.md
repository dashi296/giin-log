# web (Plan 2) Implementation Plan — 議員一覧・議員詳細

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** `@giin-log/db` のスキーマ・シードを使い、TanStack Start(Cloudflare Workers + D1)で議員一覧・議員詳細の2画面を、ローカル D1 で実起動するところまで作る。

**Architecture:** web は Feature-Sliced Design。クエリは `entities/*/api` にドライバ非依存関数(Drizzle インスタンスを引数で受ける)として集約し、テストは db と同じく better-sqlite3 + 同マイグレーション + シードで行う(ネットワーク非依存)。本番/dev の D1 アクセスは `cloudflare:workers` の `env.DB` から `drizzle-orm/d1` で生成し、routes の loader が薄く呼ぶ。UI は shadcn を独立ディレクトリに置き、`shared/ui` を唯一の窓口にする。

**Tech Stack:** TanStack Start (`@tanstack/react-start`) + TanStack Router, React 19, Vite, `@cloudflare/vite-plugin`, wrangler, Cloudflare D1, `drizzle-orm/d1` / `drizzle-orm/better-sqlite3`, Tailwind CSS v4 + shadcn/ui, vitest + @testing-library/react (jsdom), oxlint/oxfmt。

## Global Constraints

- 設計正典: `docs/superpowers/specs/2026-06-24-otsu-giin-log-design.md`。本計画のサブ設計: `docs/superpowers/specs/2026-06-25-web-design.md`。矛盾時は正典が優先。
- パッケージは ESM(`"type": "module"`)。スキーマは `@giin-log/db` の Drizzle 定義が単一情報源。**手書きの行型を作らない**(`@giin-log/db` の推論型を再export)。
- FSD 依存方向: `routes → features → entities → shared → shadcn`。逆方向・同一レイヤー依存を作らない。
- **shadcn を import できるのは `shared/ui` だけ。** entities/features/routes は `shared/ui` 経由でのみ UI を使う。shadcn は `web/src/shadcn/` の独立ディレクトリ(FSD 層に依存しない純プリミティブ)。
- **`cloudflare:workers` を import してよいのは `shared/db/get-db.ts` と routes の loader だけ。** entities・features・テスト・コンポーネントからは絶対に import しない(vitest が解決できずに壊れるため)。
- クエリ関数はドライバ非依存: 引数 `db` を `AppDb`(= `BaseSQLiteDatabase<"async", unknown, typeof schema>`)で型付けし、本体は常に `await` する。テストは better-sqlite3 の同期 Drizzle を `AppDb` にキャストして渡す(`await` は同期値を透過するため実行時安全)。
- 現任期 = `starts_on <= today` かつ(`ends_on IS NULL OR ends_on >= today`)を満たす term のうち `starts_on` 最大。該当無しは空状態(エラーにしない)。
- 透明性: 各画面に出典リンク(`source_url`)+ 最終更新日(`updated_at`)を表示。出席の出典は `meetings.source_url` を継承。
- スコープは2画面のみ(議員一覧 `/`・議員詳細 `/councilors/$slug`)。発言検索・会議一覧・クラウドデプロイは対象外。
- コミットメッセージ末尾に必ず: `Co-Authored-By: Claude Opus 4.8 (1M context) <noreply@anthropic.com>`
- push / クラウドデプロイはユーザーの明示指示を待つ(本計画はローカルのみ)。

```ts
import type { BaseSQLiteDatabase } from "drizzle-orm/sqlite-core"
```

- [ ] **Step 3: テストハーネス `makeWebTestDb` を作成**

`web/src/shared/db/testing.ts`:

```ts
import Database from "better-sqlite3"
import { drizzle } from "drizzle-orm/better-sqlite3"
import { migrate } from "drizzle-orm/better-sqlite3/migrator"
import { MIGRATIONS_FOLDER, applySeed, schema } from "@giin-log/db"
import type { AppDb } from "./types.js"

export function makeWebTestDb(seed = true): {
  db: AppDb
  sqlite: Database.Database
} {
  const sqlite = new Database(":memory:")
  sqlite.pragma("foreign_keys = ON")
  const d = drizzle(sqlite, { schema })
  migrate(d, { migrationsFolder: MIGRATIONS_FOLDER })
  if (seed) applySeed(sqlite)
  return { db: d as unknown as AppDb, sqlite }
}
```

- [ ] **Step 4: getDb を作成(cloudflare:workers 専用境界)**

`web/src/shared/db/get-db.ts`:

```ts
import { drizzle } from "drizzle-orm/d1"
import { env } from "cloudflare:workers"
import { schema } from "@giin-log/db"
import type { AppDb } from "./types.js"

// routes の loader 専用。env.DB はリクエスト内でのみ有効なので関数で生成する。
export function getDb(): AppDb {
  return drizzle(env.DB, { schema }) as unknown as AppDb
}
```

```ts
import { schema } from "@giin-log/db"
import type { Term } from "@giin-log/db"
import type { AppDb } from "@/shared/db/types"

export async function getCurrentTerm(
  db: AppDb,
  today: string,
): Promise<Term | undefined> {
  const rows = await db
    .select()
    .from(schema.terms)
    .where(
      and(
        lte(schema.terms.startsOn, today),
        or(isNull(schema.terms.endsOn), gte(schema.terms.endsOn, today)),
      ),
    )
    .orderBy(desc(schema.terms.startsOn))
    .limit(1)
    .all()
  return rows[0]
}
```

```ts
import { eq, and, asc } from "drizzle-orm"
import { schema } from "@giin-log/db"
import type { AppDb } from "@/shared/db/types"
import type { CouncilorListItem } from "../model/index.js"

export async function listCouncilors(
  db: AppDb,
  termId: number,
): Promise<CouncilorListItem[]> {
  const rows = await db
    .select({
      id: schema.councilors.id,
      slug: schema.councilors.slug,
      name: schema.councilors.name,
      nameKana: schema.councilors.nameKana,
      photoUrl: schema.councilors.photoUrl,
      faction: schema.memberships.faction,
      generalQuestionCount: schema.councilorStats.generalQuestionCount,
      honkaigiAttendanceRate: schema.councilorStats.honkaigiAttendanceRate,
      sourceUrl: schema.councilors.sourceUrl,
      updatedAt: schema.councilors.updatedAt,
    })
    .from(schema.memberships)
    .innerJoin(
      schema.councilors,
      eq(schema.councilors.id, schema.memberships.councilorId),
    )
    .leftJoin(
      schema.councilorStats,
      and(
        eq(schema.councilorStats.councilorId, schema.memberships.councilorId),
        eq(schema.councilorStats.termId, schema.memberships.termId),
      ),
    )
    .where(eq(schema.memberships.termId, termId))
    .orderBy(asc(schema.councilors.nameKana))
    .all()

  return rows.map((r) => ({
    ...r,
    generalQuestionCount: r.generalQuestionCount ?? 0,
  }))
}
```

```ts
import { eq, and, desc, asc } from "drizzle-orm"
import { schema } from "@giin-log/db"
import type { Councilor, Membership, Meeting, Statement } from "@giin-log/db"
import type { AppDb } from "@/shared/db/types"

export interface TimelineItem {
  statement: Statement
  meeting: Pick<Meeting, "id" | "name" | "heldOn" | "kind" | "sourceUrl">
}

export interface CouncilorDetail {
  councilor: Councilor
  membership: Membership | undefined
  stats:
    | {
        generalQuestionCount: number
        statementCount: number
        honkaigiAttendanceRate: number | null
      }
    | undefined
  timeline: TimelineItem[]
}

export async function getCouncilorDetail(
  db: AppDb,
  slug: string,
  termId: number,
): Promise<CouncilorDetail | undefined> {
  const councilorRows = await db
    .select()
    .from(schema.councilors)
    .where(eq(schema.councilors.slug, slug))
    .limit(1)
    .all()
  const councilor = councilorRows[0]
  if (!councilor) return undefined

  const membershipRows = await db
    .select()
    .from(schema.memberships)
    .where(
      and(
        eq(schema.memberships.councilorId, councilor.id),
        eq(schema.memberships.termId, termId),
      ),
    )
    .limit(1)
    .all()

  const statsRows = await db
    .select({
      generalQuestionCount: schema.councilorStats.generalQuestionCount,
      statementCount: schema.councilorStats.statementCount,
      honkaigiAttendanceRate: schema.councilorStats.honkaigiAttendanceRate,
    })
    .from(schema.councilorStats)
    .where(
      and(
        eq(schema.councilorStats.councilorId, councilor.id),
        eq(schema.councilorStats.termId, termId),
      ),
    )
    .limit(1)
    .all()

  const timelineRows = await db
    .select({
      statement: schema.statements,
      meeting: {
        id: schema.meetings.id,
        name: schema.meetings.name,
        heldOn: schema.meetings.heldOn,
        kind: schema.meetings.kind,
        sourceUrl: schema.meetings.sourceUrl,
      },
    })
    .from(schema.statements)
    .innerJoin(
      schema.meetings,
      eq(schema.meetings.id, schema.statements.meetingId),
    )
    .where(eq(schema.statements.councilorId, councilor.id))
    .orderBy(desc(schema.meetings.heldOn), asc(schema.statements.sequence))
    .all()

  return {
    councilor,
    membership: membershipRows[0],
    stats: statsRows[0],
    timeline: timelineRows,
  }
}
```

```ts
import type { CouncilorListItem } from "@/entities/councilor/model"

export type SortKey = "kana" | "questions" | "attendance"

export function factionsOf(items: CouncilorListItem[]): string[] {
  const set = new Set<string>()
  for (const i of items) if (i.faction) set.add(i.faction)
  return [...set].sort((a, b) => a.localeCompare(b, "ja"))
}

export function applyControls(
  items: CouncilorListItem[],
  opts: { faction: string | null; sort: SortKey },
): CouncilorListItem[] {
  const filtered = opts.faction
    ? items.filter((i) => i.faction === opts.faction)
    : items.slice()

  const byKana = (a: CouncilorListItem, b: CouncilorListItem) =>
    (a.nameKana ?? "").localeCompare(b.nameKana ?? "", "ja")

  switch (opts.sort) {
    case "questions":
      return filtered.sort(
        (a, b) => b.generalQuestionCount - a.generalQuestionCount || byKana(a, b),
      )
    case "attendance":
      return filtered.sort((a, b) => {
        const av = a.honkaigiAttendanceRate
        const bv = b.honkaigiAttendanceRate
        if (av === null && bv === null) return byKana(a, b)
        if (av === null) return 1
        if (bv === null) return -1
        return bv - av || byKana(a, b)
      })
    case "kana":
    default:
      return filtered.sort(byKana)
  }
}
```
